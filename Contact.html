/** ** Java Program to implement Knuth Morris Pratt Algorithm **/ import java.io.IOException; /** Class KnuthMorrisPratt **/ public class KMP { /** Failure array **/ /** Constructor **/ public static int[] postfix(String text, String pat) { /** pre construct failure array for a pattern **/ int[] failure = new int[pat.length()]; failure = fail(pat, failure); /** find match **/ return posMatch(text, pat, failure); } /** Failure function for a pattern **/ private static int[] fail(String pat, int[] failure) { int n = pat.length(); failure[0] = -1; for (int j = 1; j < n; j++) { int i = failure[j - 1]; while ((pat.charAt(j) != pat.charAt(i + 1)) && i >= 0) i = failure[i]; if (pat.charAt(j) == pat.charAt(i + 1)) failure[j] = i + 1; else failure[j] = -1; } return failure; } /** Function to find match for a pattern **/ private static int[] posMatch(String text, String pat, int[] failure) { int[] out = new int[3]; int i = 0, j = 0; int matchStart = -1; int matchEnd = -1; int matchStartOld = -1; int matchEndOld = -1; int oldDiff = -1; int newDiff = -1; int lens = text.length(); int lenp = pat.length(); while (i < lens && j < lenp) { if (text.charAt(i) == pat.charAt(j)) { if(matchStart == -1) { matchStart = i; } i++; j++; } else { //not the ending index :: if(matchStart != -1) { matchEnd = i; newDiff = matchEnd - matchStart; if(newDiff >= oldDiff) { matchStartOld = matchStart; matchEndOld = matchEnd; oldDiff = newDiff; } } matchStart = -1; if (j == 0) i++; else j = failure[j - 1] + 1; } if((j== lenp)) { matchEnd = i; newDiff = matchEnd - matchStart; if(newDiff > oldDiff) { matchStartOld = matchStart; matchEndOld = matchEnd; oldDiff = newDiff; //System.out.println("start : " + matchStart + ", end "+ i); } j =0; } } if(matchStart != -1) { matchEnd = i; newDiff = matchEnd - matchStart; if(newDiff >= oldDiff) { matchStartOld = matchStart; matchEndOld = matchEnd; oldDiff = newDiff; } } System.out.println("start : " + matchStartOld + ", end "+ matchEndOld); out[0] = matchStartOld; out[1] = matchEndOld; out[2] = oldDiff; return out; } /** Main Function **/ public static void main(String[] args) throws IOException { String text = "nothing"; String prefix = "bruno"; String postfix = "ingenious"; int[] pos = postfix(text, postfix); System.out.println(text.substring(pos[0], pos[1])); //for prefix set both of them reverse StringBuilder sb = new StringBuilder(text); String pre = sb.reverse().toString(); String prech = new StringBuilder(prefix).reverse().toString(); int[] preOut = postfix(pre, prech); int len = text.length(); System.out.println("first : " + (len -preOut[1])+", "+ (len -preOut[0])); System.out.println(text.substring(len -preOut[1], len- preOut[0])); } }
